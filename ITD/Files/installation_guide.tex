\section{Prerequisites}
Before proceeding, ensure the following tools are installed on your machine:

\begin{itemize}
    \item \textbf{Java Development Kit}: Version 25 or higher.
    \item \textbf{Bun}: Visit \url{https://bun.sh} for installation.
    \item \textbf{PostgreSQL}: Visit \url{https://www.postgresql.org/} for installation.
    \item \textbf{Mapbox API Key}: Visit \url{https://www.mapbox.com/} to get an API key.
\end{itemize}

\section{Backend}

\subsection{Build}
Go to the backend directory:
\begin{lstlisting}
cd implementation/backend/bbp
\end{lstlisting}

\noindent Build the executable JAR:
\begin{lstlisting}
./mvnw clean package -DskipTests
\end{lstlisting}

\subsection{Run}
Run the compiled application:
\begin{lstlisting}
java -jar target/bbp-backend.jar
\end{lstlisting}

\noindent The backend server will start on port \texttt{8080}.

\section{Frontend}

\subsection{Build}
Go to the frontend directory:
\begin{lstlisting}
cd implementation/frontend/bbp
\end{lstlisting}

\noindent Install dependencies and compile the standalone executable:
\begin{lstlisting}
bun install
bun run build:prod
\end{lstlisting}

\subsection{Run}
Run the generated binary. You have to provide your Mapbox API key:
\begin{lstlisting}
./bbp-frontend --mapbox.api.key=YOUR_KEY
\end{lstlisting}

\noindent You can also specify ports if they differ from the defaults:
\begin{lstlisting}
./bbp-frontend --mapbox.api.key=YOUR_KEY --frontend.port=3000 --backend.port=8080
\end{lstlisting}

\noindent \textbf{The application will be accessible at \url{http://localhost:3000}.}
